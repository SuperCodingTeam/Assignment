예외 처리는 panic 함수를 사용합니다. 함수는 showMenu, calculate 함수를 정의합니다.

— 태호 풀이 —

---

```go
package main

import "fmt"

func main() {
	for  {
		var x int
		var y int
		var op string

		fmt.Print("Enter first number: ")
		fmt.Scan(&x)
		fmt.Print("Enter second number: ")
		fmt.Scan(&y)
		showMenu()
		fmt.Print("Enter operator: ")
		fmt.Scan(&op)
		if (op == "q") {
			break
		}

		result := calculate(x, y, op)
		fmt.Println("Result: ", result)
	}
}

func showMenu() {
	fmt.Println("Calculator")
	fmt.Println("+")
	fmt.Println("-")
	fmt.Println("*")
	fmt.Println("/")
	fmt.Println("q(quit)")
}

func calculate(x int, y int, op string) int {
	switch op {
	case "+":
		return x + y
	case "-":
		return x - y
	case "*":
		return x * y
	case "/":
		return x / y
	case "%":
		return x % y
	default:
		panic("Invalid operator")
	}
}
```
